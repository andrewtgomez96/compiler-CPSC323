[* this is comment for this sample code which
   converts  Fahrenheit into Celcius *]

function convert1x (fahr : int)
{
           return 5 * (fahr -38.8978978909087) / 9;
}

$$
     int    low,  high, step;      [* declarations *]

     get (low, high, step);
     while (low  <  high )  
         {  put (low);
             put (convert1x  (low));
             low =  low  +  step;
         } 
      whileend

 $$
